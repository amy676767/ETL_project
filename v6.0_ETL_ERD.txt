Vehicles
-
id INT PK
year INT
model VARCHAR(255)
vehicle_type VARCHAR(255)
brand VARCHAR(255)


Sources
-
id INT PK
name VARCHAR(255) # Autotrader.com, Ebay.com, etc...


#   This inventory table will be the result of joining the following entities together: "Vehicles", "Autotrader.com", "Ebay.com"

Inventory
-
id INT PK
source_id INT FK >- Sources.id
vehicle_id INT FK >- Vehicles.id
condition VARCHAR(255)
engine VARCHAR(255)
exterior_color VARCHAR(255)
drive_type VARCHAR(255)
fuel_type VARCHAR(255)
interior_color VARCHAR(255)
make VARCHAR(255)
mileage INT
model VARCHAR(255)
price MONEY
transmission VARCHAR(255)
seller_name VARCHAR(255)
url VARCHAR(MAX)
vin VARCHAR(100)
year INT
